  <script>
    document.getElementById('calculateBtn').addEventListener('click', function() {
      const revenue = parseFloat(document.getElementById('revenue').value) || 0;
      const relocation = parseFloat(document.getElementById('relocation').value) || 0;
      const expenses = parseFloat(document.getElementById('expenses').value) || 0;
      const quality = parseFloat(document.getElementById('quality').value) || 0;
      const childcare = parseFloat(document.getElementById('childcare').value) || 0;
      const wrapCare = document.getElementById('wrapCare').value;

      // Adjust childcare multiplier based on wrap-around care availability
      const childcareAdjustment = wrapCare === 'yes' ? 0.9 : 1.1;
      const adjustedChildcare = childcare * childcareAdjustment;

      // Factor in preschool quality on total expenses (higher-quality may increase costs slightly)
      const qualityMultiplier = 1 + (quality - 5) * 0.02; // each point above/below 5 changes by 2%

      const totalMonthlyExpenses = (expenses + adjustedChildcare) * qualityMultiplier;
      const cashFlow = revenue - totalMonthlyExpenses;

      // Runway in months = Relocation budget / Abs monthly negative cash flow (if negative)
      let runwayMonths = 0;
      if (cashFlow < 0 && relocation > 0) {
        runwayMonths = Math.floor(relocation / Math.abs(cashFlow));
      } else if (cashFlow >= 0) {
        runwayMonths = Infinity;
      }

      // Insights
      let insight = '';
      if (cashFlow > 0) {
        insight = 'You’re currently operating sustainably — your relocation costs can be recovered over time.';
      } else if (runwayMonths < 6) {
        insight = '⚠️ Your runway is under 6 months. Consider reducing fixed costs or exploring childcare savings.';
      } else if (runwayMonths < 12) {
        insight = 'Moderate runway — structure growth goals for stability within the next year.';
      } else {
        insight = 'Strong runway — you have flexibility to focus on strategic relocation and scaling.';
      }

      // Update UI
      document.getElementById('totalExpenses').innerText = totalMonthlyExpenses.toFixed(2);
      document.getElementById('cashFlow').innerText = cashFlow.toFixed(2);
      document.getElementById('runwayMonths').innerText = isFinite(runwayMonths) ? runwayMonths : '∞';
      document.getElementById('insight').innerText = insight;
      document.getElementById('results').classList.remove('hidden');
    });
  </script>
